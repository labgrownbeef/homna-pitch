# 4.2 Configuration

```toml
# config/config.toml
[chat]
model = "gpt-4"
end_point = "https://api.openai.com/v1/chat/completions"
tokenization_model_name = "gpt-4"

[memory]
working_memory_span = 5  # days
short_term_memory_span = 30  # days
long_term_memory_span = 365  # days

[market_data]
global_data_provider = "yahoo_finance"
saudi_data_provider = "tadawul_api"
update_frequency = 60  # seconds

[sentiment_analysis]
global_model = "financial_sentiment_bert"
arabic_model = "arabic_financial_sentiment_bert"
social_media_sources = ["twitter", "stocktwits", "reddit"]
news_sources = ["bloomberg", "reuters", "cnbc", "argaam", "aleqtisadiah"]
```
