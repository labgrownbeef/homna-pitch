# 9.2 Trading Metrics

```cpp
class TradingMetrics {
private:
    struct LatencyHistogram {
        std::atomic<uint64_t> buckets[20];  // 0-1μs, 1-2μs, ... 19-20μs, >20μs
        
        void record(uint64_t latency_ns) {
            size_t bucket = std::min(latency_ns / 1000, size_t(19));
            buckets[bucket].fetch_add(1, std::memory_order_relaxed);
        }
    };
    
    LatencyHistogram market_data_latency;
    LatencyHistogram order_latency;
    LatencyHistogram tick_to_trade_latency;
    
    std::atomic<uint64_t> messages_processed{0};
    std::atomic<uint64_t> orders_sent{0};
    std::atomic<uint64_t> orders_filled{0};
    std::atomic<uint64_t> orders_rejected{0};
    
public:
    void record_market_data_latency(uint64_t ns) {
        market_data_latency.record(ns);
        messages_processed.fetch_add(1);
    }
    
    void record_order_latency(uint64_t ns) {
        order_latency.record(ns);
        orders_sent.fetch_add(1);
    }
    
    void generate_report() {
        log_info("Trading Metrics Report:");
        log_info("  Messages Processed: {}", messages_processed.load());
        log_info("  Orders Sent: {}", orders_sent.load());
        log_info("  Orders Filled: {}", orders_filled.load());
        log_info("  Fill Rate: {:.2f}%", 
                 (double)orders_filled / orders_sent * 100);
        
        // Latency percentiles
        log_info("  Market Data Latency:");
        print_latency_percentiles(market_data_latency);
        
        log_info("  Order Latency:");
        print_latency_percentiles(order_latency);
    }
};
```
