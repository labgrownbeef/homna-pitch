# 9.1 System Monitoring

```cpp
class SystemMonitor {
private:
    struct SystemMetrics {
        double cpu_usage[32];  // Per-core usage
        uint64_t memory_used;
        uint64_t memory_available;
        uint64_t network_rx_bytes;
        uint64_t network_tx_bytes;
        uint64_t network_rx_packets;
        uint64_t network_tx_packets;
        double disk_io_read_mbps;
        double disk_io_write_mbps;
    };
    
    SystemMetrics current_metrics;
    
public:
    void collect_metrics() {
        // CPU metrics
        for (int i = 0; i < 32; ++i) {
            current_metrics.cpu_usage[i] = get_cpu_usage(i);
        }
        
        // Memory metrics
        struct sysinfo info;
        sysinfo(&info);
        current_metrics.memory_used = info.totalram - info.freeram;
        current_metrics.memory_available = info.freeram;
        
        // Network metrics
        read_network_stats(current_metrics);
        
        // Disk I/O
        read_disk_stats(current_metrics);
    }
    
    void check_thresholds() {
        // CPU threshold
        for (int i = 2; i < 32; ++i) {  // Skip system cores 0-1
            if (current_metrics.cpu_usage[i] > 80.0) {
                alert("High CPU usage on core {}: {:.1f}%", 
                      i, current_metrics.cpu_usage[i]);
            }
        }
        
        // Memory threshold
        double memory_usage_pct = 
            (double)current_metrics.memory_used / 
            (current_metrics.memory_used + current_metrics.memory_available);
            
        if (memory_usage_pct > 0.85) {
            alert("High memory usage: {:.1f}%", memory_usage_pct * 100);
        }
    }
};
```
