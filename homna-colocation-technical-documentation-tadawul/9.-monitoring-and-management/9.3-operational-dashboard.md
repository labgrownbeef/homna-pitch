# 9.3 Operational Dashboard

```yaml
Grafana_Dashboard:
  Infrastructure:
    panels:
      - title: "CPU Usage by Core"
        query: "cpu_usage{core=~'2-31'}"
        visualization: "heatmap"
        
      - title: "Memory Usage"
        query: "memory_used / memory_total * 100"
        visualization: "gauge"
        
      - title: "Network Throughput"
        query: "rate(network_bytes[1m])"
        visualization: "graph"
        
      - title: "Disk I/O"
        query: "rate(disk_io_bytes[1m])"
        visualization: "graph"
        
  Trading:
    panels:
      - title: "Message Rate"
        query: "rate(messages_processed[1s])"
        visualization: "graph"
        
      - title: "Order Rate"
        query: "rate(orders_sent[1s])"
        visualization: "graph"
        
      - title: "Latency Heatmap"
        query: "latency_histogram"
        visualization: "heatmap"
        
      - title: "P&L"
        query: "cumulative_pnl"
        visualization: "graph"
        
  Alerts:
    - name: "High Latency"
      condition: "p99_latency > 100"
      severity: "warning"
      
    - name: "Low Fill Rate"
      condition: "fill_rate < 0.5"
      severity: "warning"
      
    - name: "System Overload"
      condition: "cpu_usage > 0.9 OR memory_usage > 0.9"
      severity: "critical"
```
