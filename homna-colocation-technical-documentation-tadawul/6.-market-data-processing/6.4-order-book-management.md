# 6.4 Order Book Management

```cpp
// Ultra-fast order book implementation
class OrderBook {
private:
    static constexpr size_t MAX_LEVELS = 100;
    
    struct PriceLevel {
        uint32_t price;
        uint32_t quantity;
        uint16_t order_count;
        uint16_t _padding;  // Alignment
    };
    
    struct alignas(64) BookSide {
        PriceLevel levels[MAX_LEVELS];
        uint32_t depth;
        
        void insert_order(uint32_t price, uint32_t quantity) noexcept;
        void remove_order(uint32_t price, uint32_t quantity) noexcept;
    };
    
    BookSide bids;
    BookSide asks;
    std::atomic<uint64_t> last_update_time;
    
public:
    void add_order(Side side, uint32_t price, uint32_t quantity) noexcept {
        if (side == Side::BID) {
            bids.insert_order(price, quantity);
        } else {
            asks.insert_order(price, quantity);
        }
        
        last_update_time.store(get_timestamp(), 
                             std::memory_order_relaxed);
    }
    
    inline std::pair<uint32_t, uint32_t> get_best_bid_ask() const noexcept {
        return {
            bids.depth > 0 ? bids.levels[0].price : 0,
            asks.depth > 0 ? asks.levels[0].price : 0
        };
    }
    
    inline uint32_t get_mid_price() const noexcept {
        auto [bid, ask] = get_best_bid_ask();
        return bid && ask ? (bid + ask) >> 1 : 0;  // Fast division by 2
    }
};
```
