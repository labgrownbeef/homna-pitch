# 6.3 GLIMPSE Recovery Protocol

```cpp
// GLIMPSE protocol for order book recovery and gap filling
namespace GLIMPSE {

#pragma pack(push, 1)
struct LoginRequest {
    char packet_type = 'L';
    char username[6];
    char password[10];
    char requested_session[10];
    char requested_sequence_number[20];
};

struct LoginAccepted {
    char packet_type = 'A';
    char session[10];
    uint64_t sequence_number;
};

struct LoginRejected {
    char packet_type = 'J';
    char reject_reason;
};

struct SnapshotRequest {
    char packet_type = 'R';
    char symbol[12];
    uint32_t source_id;
};

struct SnapshotMessage {
    char packet_type = 'S';
    uint64_t sequence_number;
    uint32_t message_count;
    // Followed by message_count ITCH messages
};
#pragma pack(pop)

class GLIMPSEClient {
private:
    // Connection parameters
    struct ConnectionConfig {
        std::string primary_host;
        uint16_t primary_port;
        std::string backup_host;
        uint16_t backup_port;
        std::string username;
        std::string password;
        std::chrono::seconds timeout{30};
        uint32_t max_retry_attempts{3};
    };
    
    ConnectionConfig config;
    int socket_fd{-1};
    bool is_connected{false};
    std::mutex connection_mutex;
    
    // Recovery state
    struct RecoveryState {
        uint64_t last_good_sequence{0};
        uint64_t target_sequence{0};
        bool recovery_in_progress{false};
        std::chrono::steady_clock::time_point recovery_start;
        std::vector<std::string> symbols_to_recover;
    };
    
    RecoveryState recovery_state;
    
public:
    GLIMPSEClient(const ConnectionConfig& cfg) : config(cfg) {}
    
    // Connect to GLIMPSE service
    bool connect() {
        std::lock_guard<std::mutex> lock(connection_mutex);
        
        // Try primary connection
        socket_fd = create_tcp_socket();
        if (connect_to_server(config.primary_host, config.primary_port)) {
            is_connected = true;
            return true;
        }
        
        // Fallback to backup
        close(socket_fd);
        socket_fd = create_tcp_socket();
        if (connect_to_server(config.backup_host, config.backup_port)) {
            is_connected = true;
            log_warning("Connected to backup GLIMPSE server");
            return true;
        }
        
        log_error("Failed to connect to GLIMPSE servers");
        return false;
    }
    
    // Login to GLIMPSE service
    bool login(const std::string& session, uint64_t start_sequence) {
        LoginRequest request;
        request.packet_type = 'L';
        
        // Fill credentials
        std::strncpy(request.username, config.username.c_str(), 
                    sizeof(request.username));
        std::strncpy(request.password, config.password.c_str(), 
                    sizeof(request.password));
        std::strncpy(request.requested_session, session.c_str(), 
                    sizeof(request.requested_session));
        
        // Convert sequence number to string
        std::snprintf(request.requested_sequence_number, 
                     sizeof(request.requested_sequence_number),
                     "%020lu", start_sequence);
        
        // Send login request
        if (!send_glimpse_message(&request, sizeof(request))) {
            return false;
        }
        
        // Wait for response
        char response_buffer[1024];
        ssize_t bytes_read = read_with_timeout(response_buffer, 
                                              sizeof(response_buffer));
        
        if (bytes_read < 1) {
            log_error("No response to GLIMPSE login");
            return false;
        }
        
        // Process response
        if (response_buffer[0] == 'A') {
            LoginAccepted* accepted = 
                reinterpret_cast<LoginAccepted*>(response_buffer);
            log_info("GLIMPSE login accepted, sequence: {}", 
                    accepted->sequence_number);
            return true;
        } else if (response_buffer[0] == 'J') {
            LoginRejected* rejected = 
                reinterpret_cast<LoginRejected*>(response_buffer);
            log_error("GLIMPSE login rejected: {}", 
                     get_reject_reason(rejected->reject_reason));
            return false;
        }
        
        return false;
    }
    
    // Request order book snapshot for recovery
    bool request_snapshot(const std::string& symbol) {
        SnapshotRequest request;
        request.packet_type = 'R';
        
        std::strncpy(request.symbol, symbol.c_str(), sizeof(request.symbol));
        request.source_id = 0;  // All sources
        
        return send_glimpse_message(&request, sizeof(request));
    }
    
    // Process snapshot response
    void process_snapshot(const char* buffer, size_t length) {
        const SnapshotMessage* snapshot = 
            reinterpret_cast<const SnapshotMessage*>(buffer);
        
        log_info("Received snapshot: {} messages starting at sequence {}", 
                snapshot->message_count, snapshot->sequence_number);
        
        // Process embedded ITCH messages
        const char* messages = buffer + sizeof(SnapshotMessage);
        size_t remaining = length - sizeof(SnapshotMessage);
        
        for (uint32_t i = 0; i < snapshot->message_count && remaining > 0; ++i) {
            // Read message length (first 2 bytes of ITCH message)
            uint16_t msg_length = ntohs(*reinterpret_cast<const uint16_t*>(messages));
            
            if (msg_length > remaining) {
                log_error("Malformed snapshot message");
                break;
            }
            
            // Process ITCH message
            process_recovered_message(messages, msg_length + 2);
            
            messages += msg_length + 2;
            remaining -= msg_length + 2;
        }
        
        // Update recovery state
        recovery_state.last_good_sequence = 
            snapshot->sequence_number + snapshot->message_count;
    }
    
    // Main recovery function
    void recover_from_gap(uint64_t gap_start, uint64_t gap_end,
                         const std::vector<std::string>& affected_symbols) {
        log_info("Starting GLIMPSE recovery for gap {}-{}", 
                gap_start, gap_end);
        
        recovery_state.last_good_sequence = gap_start - 1;
        recovery_state.target_sequence = gap_end;
        recovery_state.recovery_in_progress = true;
        recovery_state.recovery_start = std::chrono::steady_clock::now();
        recovery_state.symbols_to_recover = affected_symbols;
        
        // Connect if not already connected
        if (!is_connected && !connect()) {
            log_error("Failed to connect for recovery");
            recovery_state.recovery_in_progress = false;
            return;
        }
        
        // Login with gap start sequence
        if (!login("ITCH", gap_start)) {
            log_error("Failed to login for recovery");
            recovery_state.recovery_in_progress = false;
            return;
        }
        
        // Request snapshots for affected symbols
        for (const auto& symbol : affected_symbols) {
            if (!request_snapshot(symbol)) {
                log_error("Failed to request snapshot for {}", symbol);
            }
        }
        
        // Process recovery messages
        process_recovery_messages();
        
        // Check recovery success
        if (recovery_state.last_good_sequence >= gap_end) {
            log_info("Recovery successful, caught up to sequence {}", 
                    recovery_state.last_good_sequence);
        } else {
            log_error("Recovery incomplete, only reached sequence {}", 
                     recovery_state.last_good_sequence);
        }
        
        recovery_state.recovery_in_progress = false;
    }
    
private:
    void process_recovery_messages() {
        char buffer[65536];  // 64KB buffer
        
        while (recovery_state.recovery_in_progress && 
               recovery_state.last_good_sequence < recovery_state.target_sequence) {
            
            // Check timeout
            auto elapsed = std::chrono::steady_clock::now() - 
                          recovery_state.recovery_start;
            if (elapsed > config.timeout) {
                log_error("Recovery timeout after {}s", 
                         std::chrono::duration_cast<std::chrono::seconds>(elapsed).count());
                break;
            }
            
            // Read messages
            ssize_t bytes_read = read_with_timeout(buffer, sizeof(buffer));
            if (bytes_read <= 0) {
                continue;
            }
            
            // Process based on message type
            char msg_type = buffer[2];  // After 2-byte length
            
            switch (msg_type) {
                case 'S':  // Snapshot
                    process_snapshot(buffer + 2, bytes_read - 2);
                    break;
                    
                case 'H':  // Server heartbeat
                    // No action needed
                    break;
                    
                case 'Z':  // End of session
                    log_info("End of GLIMPSE session");
                    recovery_state.recovery_in_progress = false;
                    break;
                    
                default:
                    // Regular ITCH message
                    process_recovered_message(buffer, bytes_read);
                    break;
            }
        }
    }
    
    void process_recovered_message(const char* buffer, size_t length) {
        // Update order books with recovered message
        // This would typically call into your ITCH message processor
        
        // For now, just track sequence progress
        recovery_state.last_good_sequence++;
    }
    
    bool send_glimpse_message(const void* data, size_t length) {
        // Add SoupBinTCP framing
        uint16_t frame_length = htons(length);
        
        struct iovec iov[2] = {
            {&frame_length, sizeof(frame_length)},
            {const_cast<void*>(data), length}
        };
        
        ssize_t sent = writev(socket_fd, iov, 2);
        return sent == (sizeof(frame_length) + length);
    }
    
    ssize_t read_with_timeout(char* buffer, size_t max_length) {
        fd_set read_fds;
        FD_ZERO(&read_fds);
        FD_SET(socket_fd, &read_fds);
        
        struct timeval tv;
        tv.tv_sec = 1;  // 1 second timeout
        tv.tv_usec = 0;
        
        int ready = select(socket_fd + 1, &read_fds, nullptr, nullptr, &tv);
        if (ready > 0) {
            return recv(socket_fd, buffer, max_length, 0);
        }
        
        return 0;
    }
};

// Integration with main market data handler
class MarketDataRecoveryManager {
private:
    GLIMPSEClient glimpse_client;
    MoldUDP64Handler* mold_handler;
    
    // Gap detection
    struct GapInfo {
        uint64_t start_sequence;
        uint64_t end_sequence;
        std::chrono::steady_clock::time_point detected_time;
        bool recovery_attempted;
    };
    
    std::vector<GapInfo> detected_gaps;
    std::mutex gaps_mutex;
    
public:
    void handle_sequence_gap(uint64_t expected, uint64_t received) {
        std::lock_guard<std::mutex> lock(gaps_mutex);
        
        GapInfo gap;
        gap.start_sequence = expected;
        gap.end_sequence = received - 1;
        gap.detected_time = std::chrono::steady_clock::now();
        gap.recovery_attempted = false;
        
        detected_gaps.push_back(gap);
        
        log_warning("Gap detected: {} messages missing ({} to {})",
                   gap.end_sequence - gap.start_sequence + 1,
                   gap.start_sequence, gap.end_sequence);
        
        // Attempt immediate recovery for small gaps
        if (gap.end_sequence - gap.start_sequence < 1000) {
            initiate_recovery(gap);
        } else {
            // For large gaps, use GLIMPSE snapshots
            initiate_glimpse_recovery(gap);
        }
    }
    
    void initiate_glimpse_recovery(const GapInfo& gap) {
        // Determine affected symbols (could be all if we don't know)
        std::vector<std::string> symbols = get_active_symbols();
        
        // Start recovery in separate thread to not block main processing
        std::thread recovery_thread([this, gap, symbols]() {
            glimpse_client.recover_from_gap(
                gap.start_sequence, 
                gap.end_sequence,
                symbols
            );
        });
        
        recovery_thread.detach();
    }
    
    // Periodic gap check and recovery
    void periodic_gap_check() {
        std::lock_guard<std::mutex> lock(gaps_mutex);
        
        auto now = std::chrono::steady_clock::now();
        
        for (auto& gap : detected_gaps) {
            if (!gap.recovery_attempted) {
                auto age = now - gap.detected_time;
                
                // Retry recovery after 5 seconds
                if (age > std::chrono::seconds(5)) {
                    log_info("Retrying recovery for gap {}-{}", 
                            gap.start_sequence, gap.end_sequence);
                    initiate_glimpse_recovery(gap);
                    gap.recovery_attempted = true;
                }
            }
        }
        
        // Clean up old gaps (> 1 minute)
        detected_gaps.erase(
            std::remove_if(detected_gaps.begin(), detected_gaps.end(),
                [now](const GapInfo& gap) {
                    return (now - gap.detected_time) > std::chrono::minutes(1);
                }),
            detected_gaps.end()
        );
    }
};

} // namespace GLIMPSE
```
