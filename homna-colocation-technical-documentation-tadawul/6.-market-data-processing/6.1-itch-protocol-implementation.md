# 6.1 ITCH Protocol Implementation

```cpp
// ITCH 5.0 message processing
namespace ITCH {

#pragma pack(push, 1)
struct MessageHeader {
    uint16_t length;
    char type;
    uint64_t timestamp;
};

struct AddOrderMessage {
    MessageHeader header;
    uint64_t order_id;
    char side;
    uint32_t shares;
    char symbol[12];
    uint32_t price;
};

struct OrderExecutedMessage {
    MessageHeader header;
    uint64_t order_id;
    uint32_t executed_shares;
    uint64_t match_number;
};
#pragma pack(pop)

class ITCHProcessor {
private:
    OrderBookManager* book_manager;
    std::atomic<uint64_t> messages_processed{0};
    
public:
    void process_message(const char* buffer, size_t length) {
        const MessageHeader* header = 
            reinterpret_cast<const MessageHeader*>(buffer);
        
        switch (header->type) {
            case 'A': // Add Order
                process_add_order(buffer);
                break;
            case 'E': // Order Executed
                process_order_executed(buffer);
                break;
            case 'C': // Order Executed with Price
                process_order_executed_price(buffer);
                break;
            case 'X': // Order Cancel
                process_order_cancel(buffer);
                break;
            case 'D': // Order Delete
                process_order_delete(buffer);
                break;
            case 'U': // Order Replace
                process_order_replace(buffer);
                break;
        }
        
        messages_processed.fetch_add(1, std::memory_order_relaxed);
    }
    
private:
    void process_add_order(const char* buffer) {
        const AddOrderMessage* msg = 
            reinterpret_cast<const AddOrderMessage*>(buffer);
        
        // Update order book
        book_manager->add_order(
            msg->symbol,
            msg->order_id,
            msg->side == 'B' ? Side::BID : Side::ASK,
            msg->price,
            msg->shares
        );
    }
};

} // namespace ITCH
```
