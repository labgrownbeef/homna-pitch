# 6.2 MoldUDP64 Implementation

```cpp
// MoldUDP64 protocol for reliable multicast
class MoldUDP64Handler {
private:
    struct MoldHeader {
        char session[10];
        uint64_t sequence_number;
        uint16_t message_count;
    };
    
    uint64_t expected_sequence;
    std::unordered_map<uint64_t, std::vector<char>> out_of_order_cache;
    
public:
    void handle_packet(const char* packet, size_t length) {
        const MoldHeader* header = 
            reinterpret_cast<const MoldHeader*>(packet);
        
        // Check sequence
        if (header->sequence_number == expected_sequence) {
            // Process in-order packet
            process_messages(packet + sizeof(MoldHeader), 
                           header->message_count);
            expected_sequence += header->message_count;
            
            // Check cache for next messages
            process_cached_messages();
        } else if (header->sequence_number > expected_sequence) {
            // Cache out-of-order packet
            cache_packet(header->sequence_number, packet, length);
            
            // Request missing packets
            request_retransmission(expected_sequence, 
                                 header->sequence_number - 1);
        }
        // Ignore old packets (sequence < expected)
    }
    
private:
    void request_retransmission(uint64_t start, uint64_t end) {
        // Send retransmission request to unicast re-request server
        RetransmitRequest request;
        request.session = current_session;
        request.start_sequence = start;
        request.end_sequence = end;
        
        send_unicast_request(&request, sizeof(request));
    }
};
```
