# 8.2 Application Security

```cpp
class SecurityManager {
private:
    struct SecurityPolicy {
        uint32_t max_order_value;
        uint32_t max_position_value;
        uint32_t max_daily_volume;
        std::set<std::string> allowed_symbols;
        std::set<std::string> restricted_accounts;
    };
    
    SecurityPolicy policy;
    
public:
    bool validate_order_security(const Order& order) {
        // Check order value
        uint64_t order_value = order.quantity * order.price;
        if (order_value > policy.max_order_value) {
            log_security_event("Order value exceeds limit", order);
            return false;
        }
        
        // Check symbol restrictions
        if (policy.allowed_symbols.find(order.symbol) == 
            policy.allowed_symbols.end()) {
            log_security_event("Symbol not allowed", order);
            return false;
        }
        
        // Check account restrictions
        if (policy.restricted_accounts.find(order.account) != 
            policy.restricted_accounts.end()) {
            log_security_event("Account restricted", order);
            return false;
        }
        
        return true;
    }
    
    void audit_trail(const Order& order, const Execution& exec) {
        // Create tamper-proof audit record
        AuditRecord record;
        record.timestamp = get_timestamp();
        record.order_details = order;
        record.execution_details = exec;
        record.hash = calculate_hash(record);
        
        // Store in append-only log
        audit_log.append(record);
    }
};
```
