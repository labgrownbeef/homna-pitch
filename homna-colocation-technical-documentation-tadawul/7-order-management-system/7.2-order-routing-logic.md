# 7.2 Order Routing Logic

```cpp
class OrderRouter {
private:
    struct RouteConfig {
        bool use_ouch;
        bool use_fix;
        uint32_t max_order_rate;
        uint32_t max_order_size;
    };
    
    RouteConfig config;
    RateLimiter rate_limiter;
    OUCHClient ouch_client;
    FIXClient fix_client;
    
public:
    void route_order(const Order& order) {
        // Pre-trade checks
        if (!validate_order(order)) {
            reject_order(order, "Failed validation");
            return;
        }
        
        // Rate limiting
        if (!rate_limiter.try_acquire()) {
            reject_order(order, "Rate limit exceeded");
            return;
        }
        
        // Route selection
        if (config.use_ouch && is_ouch_supported(order.symbol)) {
            route_via_ouch(order);
        } else if (config.use_fix) {
            route_via_fix(order);
        } else {
            reject_order(order, "No valid route");
        }
    }
    
private:
    bool validate_order(const Order& order) {
        // Size limits
        if (order.quantity > config.max_order_size) {
            return false;
        }
        
        // Price validation
        if (order.price <= 0 || order.price > MAX_PRICE) {
            return false;
        }
        
        // Symbol validation
        if (!is_valid_symbol(order.symbol)) {
            return false;
        }
        
        return true;
    }
};
```
