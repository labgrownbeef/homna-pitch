# 7.1 OUCH Protocol Implementation

```cpp
// OUCH 4.2 protocol for order entry
namespace OUCH {

#pragma pack(push, 1)
struct EnterOrderMessage {
    char type = 'O';
    uint32_t order_token;
    char buy_sell_indicator;
    uint32_t shares;
    char symbol[8];
    uint32_t price;
    uint32_t time_in_force;
    char firm[4];
    char display;
    uint64_t client_reference;
};

struct CancelOrderMessage {
    char type = 'X';
    uint32_t order_token;
    uint32_t shares;  // 0 for full cancel
};

struct ReplaceOrderMessage {
    char type = 'U';
    uint32_t existing_order_token;
    uint32_t replacement_order_token;
    uint32_t shares;
    uint32_t price;
};
#pragma pack(pop)

class OUCHClient {
private:
    int socket_fd;
    std::atomic<uint32_t> order_token_generator{1};
    std::unordered_map<uint32_t, OrderState> active_orders;
    
public:
    uint32_t send_order(const Order& order) {
        EnterOrderMessage msg;
        msg.order_token = order_token_generator.fetch_add(1);
        msg.buy_sell_indicator = order.side == Side::BUY ? 'B' : 'S';
        msg.shares = order.quantity;
        memcpy(msg.symbol, order.symbol.c_str(), 
               std::min(order.symbol.length(), sizeof(msg.symbol)));
        msg.price = order.price;
        msg.time_in_force = order.time_in_force;
        memcpy(msg.firm, "HOMN", 4);
        msg.display = 'N';  // Non-displayed
        msg.client_reference = order.client_id;
        
        send_message(&msg, sizeof(msg));
        
        // Track order state
        active_orders[msg.order_token] = OrderState::PENDING;
        
        return msg.order_token;
    }
    
    void cancel_order(uint32_t order_token, uint32_t shares = 0) {
        CancelOrderMessage msg;
        msg.order_token = order_token;
        msg.shares = shares;  // 0 for full cancel
        
        send_message(&msg, sizeof(msg));
    }
    
private:
    void send_message(const void* msg, size_t length) {
        // Add SoupBinTCP header
        uint16_t packet_length = htons(length + 1);  // +1 for packet type
        char packet_type = 'U';  // Unsequenced data
        
        struct iovec iov[3] = {
            {&packet_length, sizeof(packet_length)},
            {&packet_type, sizeof(packet_type)},
            {const_cast<void*>(msg), length}
        };
        
        writev(socket_fd, iov, 3);
    }
};

} // namespace OUCH
```
