# 4.2 Memory Management



```cpp
// Custom memory allocator for ultra-low latency
template<typename T, size_t PoolSize = 4096>
class LatencyOptimizedAllocator {
private:
    struct alignas(64) Block {
        union {
            T object;
            Block* next;
        };
    };
    
    Block pool[PoolSize];
    Block* free_list;
    std::atomic<uint32_t> allocated{0};
    
public:
    LatencyOptimizedAllocator() {
        // Initialize free list
        for (size_t i = 0; i < PoolSize - 1; ++i) {
            pool[i].next = &pool[i + 1];
        }
        pool[PoolSize - 1].next = nullptr;
        free_list = &pool[0];
    }
    
    [[nodiscard]] T* allocate() noexcept {
        if (!free_list) return nullptr;
        
        Block* block = free_list;
        free_list = free_list->next;
        allocated.fetch_add(1, std::memory_order_relaxed);
        
        return &block->object;
    }
    
    void deallocate(T* ptr) noexcept {
        if (!ptr) return;
        
        Block* block = reinterpret_cast<Block*>(ptr);
        block->next = free_list;
        free_list = block;
        allocated.fetch_sub(1, std::memory_order_relaxed);
    }
};
```
