# 4.1 Core Trading Engine Architecture

```cpp
// Low-latency trading engine structure
namespace HOMNA {

class TradingEngine {
private:
    // Lock-free components
    struct alignas(64) MarketDataProcessor {
        std::atomic<uint64_t> sequence_number{0};
        OrderBook order_books[MAX_SYMBOLS];
        std::atomic<bool> is_active{true};
        
        void process_itch_message(const ITCHMessage& msg) noexcept;
    };
    
    struct alignas(64) StrategyEngine {
        Strategy* strategies[MAX_STRATEGIES];
        std::atomic<uint32_t> active_strategies{0};
        
        void on_book_update(Symbol symbol, const OrderBook& book) noexcept;
    };
    
    struct alignas(64) OrderManager {
        OrderPool order_pool;
        std::atomic<uint64_t> order_id{1};
        
        void send_order(const Order& order) noexcept;
    };
    
public:
    void run() {
        // Pin threads to specific CPU cores
        pin_thread_to_core(std::this_thread::get_id(), 2);
        
        // Main event loop
        while (running) {
            // Process market data
            if (market_data_ready()) {
                auto msg = read_market_data();
                market_data_processor.process_itch_message(msg);
            }
            
            // Check for strategy signals
            if (signal_available()) {
                auto signal = get_signal();
                process_signal(signal);
            }
        }
    }
};

} // namespace HOMNA
```
