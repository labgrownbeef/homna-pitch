class AggressiveMarketMaker:
    def __init__(self, config):
        self.alpha_threshold = config['alpha_threshold']
        self.max_trade_size = config['max_trade_size']
        self.signal_weights = config['signal_weights']
        
    def calculate_alpha(self, market_data, signals):
        """Calculate expected short-term alpha based on market signals."""
        weighted_alpha = 0
        for signal_name, signal_value in signals.items():
            if signal_name in self.signal_weights:
                weighted_alpha += signal_value * self.signal_weights[signal_name]
                
        return weighted_alpha
        
    def generate_orders(self, market_data, signals):
        """Generate aggressive orders based on alpha signals."""
        alpha = self.calculate_alpha(market_data, signals)
        
        if abs(alpha) < self.alpha_threshold:
            return None  # No trading opportunity
            
        # Determine direction and size
        direction = 1 if alpha > 0 else -1
        size = min(self.max_trade_size, market_data.available_liquidity * abs(alpha) / self.alpha_threshold)
        
        if direction > 0:
            # Buy order
            return {
                'side': 'buy',
                'price': market_data.best_ask,
                'size': size,
                'expected_alpha': alpha
            }
        else:
            # Sell order
            return {
                'side': 'sell',
                'price': market_data.best_bid,
                'size': size,
                'expected_alpha': alpha
            }
