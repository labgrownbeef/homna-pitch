class AntiManipulationSystem {
private:
    struct Pattern {
        std::string name;
        std::function<bool(const OrderFlow&)> detector;
        double confidence_threshold;
        Action action;
    };
    
    std::vector<Pattern> manipulation_patterns = {
        {
            "Layering",
            [](const OrderFlow& flow) { return detect_layering(flow); },
            0.8,
            Action::BLOCK_AND_REPORT
        },
        {
            "Spoofing", 
            [](const OrderFlow& flow) { return detect_spoofing(flow); },
            0.9,
            Action::BLOCK_AND_REPORT
        },
        {
            "Momentum Ignition",
            [](const OrderFlow& flow) { return detect_momentum_ignition(flow); },
            0.7,
            Action::WARN_AND_MONITOR
        },
        {
            "Quote Stuffing",
            [](const OrderFlow& flow) { return detect_quote_stuffing(flow); },
            0.85,
            Action::RATE_LIMIT
        }
    };
    
public:
    bool validate_order_flow(const OrderFlow& flow) {
        for (const auto& pattern : manipulation_patterns) {
            if (pattern.detector(flow)) {
                handle_violation(pattern.name, pattern.action);
                return false;
            }
        }
        return true;
    }
    
private:
    static bool detect_layering(const OrderFlow& flow) {
        // Check for multiple orders away from market
        uint32_t away_orders = 0;
        uint32_t near_orders = 0;
        
        for (const auto& order : flow.orders) {
            double distance_pct = std::abs(order.price - flow.mid_price) / 
                                 flow.mid_price;
            
            if (distance_pct > 0.02) {  // 2% away
                away_orders++;
            } else {
                near_orders++;
            }
        }
        
        // Layering pattern: many orders away, few near
        return (away_orders > 10 && 
                away_orders > near_orders * 5 &&
                flow.cancel_rate > 0.9);
    }
    
    static bool detect_spoofing(const OrderFlow& flow) {
        // Large orders that get cancelled quickly
        for (const auto& order : flow.cancelled_orders) {
            if (order.size > flow.avg_order_size * 10 &&
                order.lifetime < std::chrono::seconds(2)) {
                return true;
            }
        }
        return false;
    }
};

class ShariahComplianceFilter:
    def __init__(self):
        self.screening_criteria = {
            'debt_ratio': 0.33,          # Max 33% debt/assets
            'interest_income': 0.05,     # Max 5% from interest
            'illiquid_assets': 0.33,     # Max 33% illiquid assets
            'prohibited_income': 0.05    # Max 5% from prohibited activities
        }
        
        self.prohibited_sectors = [
            'conventional_banking',
            'insurance',
            'alcohol',
            'gambling',
            'pork',
            'adult_entertainment',
            'weapons',
            'tobacco'
        ]
        
    def is_shariah_compliant(self, symbol):
        """
        Check if a stock is Shariah compliant for trading
        """
        company = self.get_company_data(symbol)
        
        # Sector screening
        if company.sector in self.prohibited_sectors:
            return False, "Prohibited sector"
            
        # Financial screening
        if company.debt_ratio > self.screening_criteria['debt_ratio']:
            return False, f"Debt ratio {company.debt_ratio} exceeds limit"
            
        if company.interest_income_ratio > self.screening_criteria['interest_income']:
            return False, f"Interest income {company.interest_income_ratio} exceeds limit"
            
        # Purification calculation
        purification_amount = self.calculate_purification(company)
        
        return True, {
            'compliant': True,
            'purification_required': purification_amount > 0,
            'purification_amount': purification_amount
        }
