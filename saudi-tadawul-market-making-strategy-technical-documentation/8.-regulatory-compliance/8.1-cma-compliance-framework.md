# 8.1 CMA Compliance Framework

```python
class CMAComplianceManager:
    def __init__(self):
        self.regulations = {
            'market_maker_obligations': {
                'min_presence': 0.75,  # 75% of trading day
                'max_spread': 500,     # 5% max spread
                'min_size': 1000,      # Minimum quote size
                'response_time': 5     # Seconds to requote
            },
            'position_limits': {
                'max_ownership': 0.05,  # 5% of company
                'max_daily_volume': 0.25,  # 25% of ADV
                'foreign_ownership': 0.49  # 49% for foreign investors
            },
            'reporting_requirements': {
                'daily_activity': True,
                'large_positions': True,
                'suspicious_activity': True
            }
        }
        
    def validate_market_making_activity(self, daily_activity):
        """
        Ensure compliance with CMA market making rules
        """
        violations = []
        
        # Check presence requirement
        presence = daily_activity['minutes_quoted'] / 300  # 5 hour day
        if presence < self.regulations['market_maker_obligations']['min_presence']:
            violations.append({
                'type': 'INSUFFICIENT_PRESENCE',
                'actual': presence,
                'required': self.regulations['market_maker_obligations']['min_presence']
            })
            
        # Check spread compliance
        avg_spread = daily_activity['avg_spread_bps']
        if avg_spread > self.regulations['market_maker_obligations']['max_spread']:
            violations.append({
                'type': 'EXCESSIVE_SPREAD',
                'actual': avg_spread,
                'maximum': self.regulations['market_maker_obligations']['max_spread']
            })
            
        return violations
        
    def generate_regulatory_reports(self):
        """
        Generate required CMA reports
        """
        reports = {
            'daily_market_making_report': self.generate_mm_report(),
            'position_report': self.generate_position_report(),
            'order_to_trade_ratio': self.calculate_otr(),
            'market_quality_metrics': self.calculate_market_quality()
        }
        
        return reports
```
