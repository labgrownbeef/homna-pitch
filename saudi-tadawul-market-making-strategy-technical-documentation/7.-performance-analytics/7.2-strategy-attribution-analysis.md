# 7.2 Strategy Attribution Analysis

```cpp
class StrategyAttribution {
private:
    struct StrategyPerformance {
        double pnl;
        uint64_t trade_count;
        double sharpe_ratio;
        double max_drawdown;
        double avg_holding_time;
        std::map<std::string, double> factor_attribution;
    };
    
    std::map<std::string, StrategyPerformance> strategy_performance;
    
public:
    void attribute_performance(const std::vector<Trade>& trades) {
        // Group trades by strategy
        std::map<std::string, std::vector<Trade>> strategy_trades;
        
        for (const auto& trade : trades) {
            strategy_trades[trade.strategy].push_back(trade);
        }
        
        // Calculate performance for each strategy
        for (const auto& [strategy, trades] : strategy_trades) {
            StrategyPerformance perf;
            
            // Basic metrics
            perf.pnl = std::accumulate(trades.begin(), trades.end(), 0.0,
                [](double sum, const Trade& t) { return sum + t.pnl; });
            perf.trade_count = trades.size();
            
            // Sharpe ratio
            std::vector<double> returns;
            for (const auto& trade : trades) {
                returns.push_back(trade.pnl / trade.notional);
            }
            perf.sharpe_ratio = calculate_sharpe(returns);
            
            // Factor attribution for Saudi market
            perf.factor_attribution["spread_capture"] = 
                attribute_to_spread(trades);
            perf.factor_attribution["inventory_management"] = 
                attribute_to_inventory(trades);
            perf.factor_attribution["market_timing"] = 
                attribute_to_timing(trades);
            perf.factor_attribution["prayer_time_alpha"] = 
                attribute_to_prayer_times(trades);
            
            strategy_performance[strategy] = perf;
        }
    }
    
private:
    double attribute_to_prayer_times(const std::vector<Trade>& trades) {
        double prayer_pnl = 0.0;
        double normal_pnl = 0.0;
        uint32_t prayer_count = 0;
        uint32_t normal_count = 0;
        
        for (const auto& trade : trades) {
            if (is_prayer_time(trade.timestamp)) {
                prayer_pnl += trade.pnl;
                prayer_count++;
            } else {
                normal_pnl += trade.pnl;
                normal_count++;
            }
        }
        
        // Calculate excess return during prayer times
        double prayer_avg = prayer_pnl / prayer_count;
        double normal_avg = normal_pnl / normal_count;
        
        return (prayer_avg - normal_avg) * prayer_count;
    }
};
```
