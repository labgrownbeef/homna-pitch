# 7.1 Real-Time Performance Tracking

```python
class PerformanceAnalytics:
    def __init__(self):
        self.metrics = {
            'pnl': {'realized': 0, 'unrealized': 0, 'total': 0},
            'sharpe_ratio': 0,
            'win_rate': 0,
            'profit_factor': 0,
            'max_drawdown': 0,
            'spread_capture_rate': 0,
            'inventory_turnover': 0
        }
        
    def calculate_real_time_metrics(self, trades, positions):
        """
        Calculate performance metrics in real-time
        """
        # P&L Calculation
        self.metrics['pnl']['realized'] = sum(t.pnl for t in trades)
        self.metrics['pnl']['unrealized'] = sum(
            p.unrealized_pnl for p in positions
        )
        self.metrics['pnl']['total'] = (
            self.metrics['pnl']['realized'] + 
            self.metrics['pnl']['unrealized']
        )
        
        # Win Rate
        winning_trades = [t for t in trades if t.pnl > 0]
        self.metrics['win_rate'] = len(winning_trades) / len(trades)
        
        # Profit Factor
        gross_profits = sum(t.pnl for t in trades if t.pnl > 0)
        gross_losses = abs(sum(t.pnl for t in trades if t.pnl < 0))
        self.metrics['profit_factor'] = (
            gross_profits / gross_losses if gross_losses > 0 else float('inf')
        )
        
        # Sharpe Ratio (annualized)
        returns = pd.Series([t.pnl / t.value for t in trades])
        if len(returns) > 30:
            daily_sharpe = returns.mean() / returns.std()
            self.metrics['sharpe_ratio'] = daily_sharpe * np.sqrt(252)
        
        # Spread Capture Rate
        spread_captures = [t for t in trades if t.type == 'SPREAD_CAPTURE']
        total_spreads = [t for t in trades if t.type == 'MARKET_MAKER']
        self.metrics['spread_capture_rate'] = (
            len(spread_captures) / len(total_spreads)
        )
        
        # Saudi-specific: Prayer time performance
        prayer_trades = [
            t for t in trades 
            if self.is_prayer_time(t.timestamp)
        ]
        self.metrics['prayer_time_performance'] = {
            'trades': len(prayer_trades),
            'win_rate': len([t for t in prayer_trades if t.pnl > 0]) / 
                       len(prayer_trades) if prayer_trades else 0,
            'avg_pnl': np.mean([t.pnl for t in prayer_trades]) 
                      if prayer_trades else 0
        }
        
        return self.metrics
```
