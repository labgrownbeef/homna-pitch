# 1.3 Order Book Microstructure

```cpp
class SaudiOrderBookAnalyzer {
private:
    struct OrderBookMetrics {
        double avg_spread_bps;
        double bid_depth_sar;
        double ask_depth_sar;
        double order_persistence_seconds;
        double price_impact_coefficient;
        uint32_t avg_orders_per_level;
    };
    
public:
    OrderBookMetrics analyze_book_structure(const Symbol& symbol) {
        OrderBookMetrics metrics;
        
        // Saudi stocks show distinct patterns by market cap
        if (is_large_cap(symbol)) {
            metrics.avg_spread_bps = 50;      // 0.5%
            metrics.bid_depth_sar = 2000000;  // 2M SAR
            metrics.order_persistence_seconds = 180;
            metrics.price_impact_coefficient = 0.15;
        } else if (is_mid_cap(symbol)) {
            metrics.avg_spread_bps = 150;     // 1.5%
            metrics.bid_depth_sar = 500000;   // 500K SAR
            metrics.order_persistence_seconds = 300;
            metrics.price_impact_coefficient = 0.35;
        } else {  // Small cap
            metrics.avg_spread_bps = 300;     // 3.0%
            metrics.bid_depth_sar = 100000;   // 100K SAR
            metrics.order_persistence_seconds = 600;
            metrics.price_impact_coefficient = 0.75;
        }
        
        return metrics;
    }
    
    double calculate_queue_value(const Symbol& symbol, uint32_t position) {
        // Queue position more valuable in Saudi due to persistence
        double base_value = 1.0;
        double position_factor = std::exp(-position / 5.0);  // Exponential decay
        
        // Saudi-specific: First 5 positions capture 80% of fills
        return base_value * position_factor * get_fill_probability(symbol);
    }
};
```
