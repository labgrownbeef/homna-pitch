class MarketRiskManager:
    def __init__(self):
        self.risk_params = {
            'var_confidence': 0.95,
            'var_horizon_days': 1,
            'max_portfolio_var': 0.03,      # 3% daily VaR
            'max_strategy_var': 0.02,       # 2% per strategy
            'correlation_window_days': 60,
            'stress_test_scenarios': 10
        }
        
    def calculate_portfolio_var(self, positions, market_data):
        """
        Calculate Value at Risk for the portfolio
        """
        # Get position values
        position_values = np.array([
            pos.current_shares * pos.current_price 
            for pos in positions
        ])
        
        # Get historical returns
        returns = self.get_historical_returns(
            [pos.symbol for pos in positions],
            self.risk_params['correlation_window_days']
        )
        
        # Calculate correlation matrix
        corr_matrix = returns.corr()
        
        # Calculate portfolio volatility
        weights = position_values / position_values.sum()
        portfolio_vol = np.sqrt(
            weights @ corr_matrix @ weights.T
        ) * returns.std().mean()
        
        # Calculate VaR
        z_score = stats.norm.ppf(self.risk_params['var_confidence'])
        var_pct = portfolio_vol * z_score * np.sqrt(
            self.risk_params['var_horizon_days']
        )
        
        return var_pct
        
    def perform_stress_tests(self, portfolio):
        """
        Saudi-specific stress test scenarios
        """
        scenarios = [
            {
                'name': 'Oil Price Crash',
                'shocks': {
                    'Energy': -0.15,
                    'Petrochemical': -0.12,
                    'Banks': -0.08,
                    'market_vol': 2.5
                }
            },
            {
                'name': 'Currency Devaluation',
                'shocks': {
                    'all_sectors': -0.10,
                    'Importers': -0.05,
                    'fx_rate': 0.05
                }
            },
            {
                'name': 'Regional Conflict',
                'shocks': {
                    'all_sectors': -0.12,
                    'Defense': 0.05,
                    'market_vol': 3.0,
                    'liquidity': -0.50
                }
            },
            {
                'name': 'Regulatory Change',
                'shocks': {
                    'Banks': -0.06,
                    'Insurance': -0.08,
                    'foreign_ownership': -0.20
                }
            }
        ]
        
        results = []
        for scenario in scenarios:
            pnl_impact = self.apply_scenario(portfolio, scenario)
            results.append({
                'scenario': scenario['name'],
                'pnl_impact': pnl_impact,
                'var_impact': pnl_impact / self.calculate_portfolio_value(portfolio)
            })
            
        return results
