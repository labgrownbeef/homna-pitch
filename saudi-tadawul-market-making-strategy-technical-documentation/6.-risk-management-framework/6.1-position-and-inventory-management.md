# 6.1 Position and Inventory Management

```cpp
class PositionRiskManager {
private:
    struct PositionLimits {
        // Single stock limits
        int64_t max_position_shares = 100000;
        double max_position_value = 5000000;    // 5M SAR
        double max_position_pct = 0.15;         // 15% of capital
        
        // Portfolio limits  
        double max_gross_exposure = 3.0;        // 3x leverage
        double max_net_exposure = 1.5;          // 1.5x
        
        // Sector concentration
        double max_sector_exposure = 0.40;      // 40% in one sector
        
        // Time-based limits
        int64_t max_intraday_shares = 500000;
        double max_minute_volume_pct = 0.05;    // 5% of minute volume
    };
    
    struct PositionMetrics {
        int64_t current_position;
        double position_value;
        double unrealized_pnl;
        double realized_pnl;
        std::chrono::time_point<std::chrono::steady_clock> entry_time;
        double vwap;
        double position_var;
    };
    
    PositionLimits limits;
    std::map<Symbol, PositionMetrics> positions;
    
public:
    bool validate_order(const Order& order) {
        auto& pos = positions[order.symbol];
        
        // Calculate post-trade position
        int64_t post_position = pos.current_position;
        if (order.side == Side::BUY) {
            post_position += order.size;
        } else {
            post_position -= order.size;
        }
        
        // Check absolute position limit
        if (std::abs(post_position) > limits.max_position_shares) {
            return reject_order("Position limit exceeded");
        }
        
        // Check value limit
        double post_value = std::abs(post_position) * order.price;
        if (post_value > limits.max_position_value) {
            return reject_order("Position value limit exceeded");
        }
        
        // Check portfolio percentage
        double capital = get_total_capital();
        if (post_value / capital > limits.max_position_pct) {
            return reject_order("Portfolio concentration limit exceeded");
        }
        
        // Saudi-specific: Check prayer time restrictions
        if (is_prayer_time() && !is_reducing_position(order, pos)) {
            return reject_order("Cannot increase position during prayer time");
        }
        
        return true;
    }
    
    void monitor_positions() {
        for (auto& [symbol, pos] : positions) {
            // Check position age
            auto age = std::chrono::steady_clock::now() - pos.entry_time;
            
            // Force reduce old positions
            if (age > std::chrono::hours(2)) {
                initiate_position_reduction(symbol, 0.5);  // Reduce by 50%
            }
            
            // Check unrealized P&L
            double current_price = get_current_price(symbol);
            pos.unrealized_pnl = (current_price - pos.vwap) * 
                                pos.current_position;
            
            // Stop loss at -2%
            if (pos.unrealized_pnl < -pos.position_value * 0.02) {
                initiate_stop_loss(symbol);
            }
        }
    }
};
```
