# 6.3 Operational Risk Controls

```cpp
class OperationalRiskController {
private:
    struct SystemLimits {
        // Message rate limits
        uint32_t max_orders_per_second = 100;
        uint32_t max_cancels_per_second = 200;
        uint32_t max_messages_per_second = 500;
        
        // Error thresholds
        double max_reject_rate = 0.05;          // 5% rejects
        double max_technical_error_rate = 0.01; // 1% tech errors
        
        // Latency thresholds
        uint64_t max_tick_to_trade_ns = 100000;  // 100 microseconds
        uint64_t latency_spike_threshold_ns = 500000;  // 500 microseconds
        
        // Saudi-specific
        uint32_t prayer_time_order_limit = 20;  // Reduced during prayers
        double thursday_risk_multiplier = 0.7;  // Reduce risk on last day
    };
    
    SystemLimits limits;
    RateLimiter order_rate_limiter;
    ErrorTracker error_tracker;
    LatencyMonitor latency_monitor;
    
public:
    bool pre_order_checks(const Order& order) {
        // Rate limiting
        if (!order_rate_limiter.try_acquire()) {
            return reject("Order rate limit exceeded");
        }
        
        // Check system health
        if (error_tracker.get_reject_rate() > limits.max_reject_rate) {
            return reject("High reject rate - reducing activity");
        }
        
        // Latency checks
        if (latency_monitor.get_p99() > limits.max_tick_to_trade_ns) {
            return reject("System latency too high");
        }
        
        // Saudi-specific checks
        if (is_prayer_time()) {
            if (get_recent_order_count() > limits.prayer_time_order_limit) {
                return reject("Prayer time order limit reached");
            }
        }
        
        if (is_thursday()) {
            // Apply stricter limits on Thursday (last trading day)
            if (order.value > get_normal_limit() * 
                             limits.thursday_risk_multiplier) {
                return reject("Thursday risk limit exceeded");
            }
        }
        
        return true;
    }
    
    void monitor_system_health() {
        // Continuous monitoring loop
        while (running) {
            // Check latency
            auto current_latency = latency_monitor.get_current();
            if (current_latency > limits.latency_spike_threshold_ns) {
                alert("Latency spike detected: {} ns", current_latency);
                reduce_trading_activity(0.5);  // Reduce by 50%
            }
            
            // Check error rates
            if (error_tracker.get_error_rate() > 
                limits.max_technical_error_rate) {
                alert("High error rate detected");
                pause_trading();
                diagnose_errors();
            }
            
            // Check connection health
            if (!check_exchange_connectivity()) {
                alert("Exchange connectivity issue");
                failover_to_backup();
            }
            
            std::this_thread::sleep_for(std::chrono::milliseconds(100));
        }
    }
};
```
