# 4.3 Aggressive Market Making

```cpp
class AggressiveMarketMaker {
private:
    struct AlphaSignal {
        double strength;      // -1 to 1
        double confidence;    // 0 to 1
        SignalType type;
        uint64_t timestamp;
    };
    
    struct AggressiveConfig {
        double min_alpha_threshold = 0.02;   // 2% expected move
        double max_cross_spread_pct = 0.5;   // Cross up to 50% of spread
        uint32_t max_aggressive_size = 50000;
        double signal_decay_seconds = 5.0;
    };
    
    AggressiveConfig config;
    SignalGenerator signal_generator;
    
public:
    std::vector<Order> generate_aggressive_orders(
        const Symbol& symbol,
        const OrderBook& book,
        const MarketContext& context) {
        
        std::vector<Order> orders;
        
        // Generate alpha signals
        auto signals = signal_generator.generate_signals(symbol, context);
        
        // Filter and rank signals
        auto actionable_signals = filter_signals(signals);
        
        for (const auto& signal : actionable_signals) {
            if (should_cross_spread(signal, book)) {
                // Generate aggressive order
                Order order = create_aggressive_order(signal, book);
                
                // Saudi-specific validations
                if (validate_aggressive_order(order, context)) {
                    orders.push_back(order);
                }
            }
        }
        
        return orders;
    }
    
private:
    bool should_cross_spread(const AlphaSignal& signal,
                           const OrderBook& book) {
        // Check signal strength
        if (std::abs(signal.strength) < config.min_alpha_threshold) {
            return false;
        }
        
        // Check spread crossing economics
        double spread_bps = book.get_spread_bps();
        double expected_move_bps = signal.strength * 10000;  // Convert to bps
        
        // Saudi market: wider spreads make crossing more attractive
        double cross_threshold = spread_bps * config.max_cross_spread_pct;
        
        return expected_move_bps > cross_threshold;
    }
    
    Order create_aggressive_order(const AlphaSignal& signal,
                                const OrderBook& book) {
        Order order;
        order.symbol = book.symbol;
        order.type = OrderType::LIMIT;
        
        if (signal.strength > 0) {
            // Bullish signal - buy aggressively
            order.side = Side::BUY;
            order.price = book.best_ask;  // Cross the spread
            order.size = calculate_aggressive_size(signal, book.ask_depth);
        } else {
            // Bearish signal - sell aggressively
            order.side = Side::SELL;
            order.price = book.best_bid;  // Cross the spread
            order.size = calculate_aggressive_size(signal, book.bid_depth);
        }
        
        order.time_in_force = TimeInForce::IOC;  // Immediate or cancel
        
        return order;
    }
    
    uint32_t calculate_aggressive_size(const AlphaSignal& signal,
                                     uint32_t available_depth) {
        // Base size on signal confidence
        uint32_t base_size = config.max_aggressive_size * signal.confidence;
        
        // Don't exceed available liquidity
        uint32_t size = std::min(base_size, available_depth * 0.3);
        
        // Saudi market: respect minimum trading units
        return round_to_lot_size(size);
    }
};
```
