# 4.2 Inventory-Based Market Making

```cpp
class InventoryBasedMarketMaker {
private:
    struct InventoryParams {
        // Position limits
        int64_t max_position = 1000000;     // SAR value
        int64_t target_position = 0;        // Neutral target
        double position_limit_pct = 0.15;   // 15% of capital
        
        // Skew parameters
        double inventory_skew_factor = 0.002;  // Skew per 1% inventory
        double max_skew_bps = 100;            // Max 1% skew
        
        // Mean reversion
        double mean_reversion_speed = 0.1;   // 10% per hour
        double half_life_minutes = 45;        // Saudi market half-life
    };
    
    struct RiskMetrics {
        double current_var;
        double max_var;
        double inventory_risk;
        double market_risk;
    };
    
    InventoryParams params;
    std::map<Symbol, int64_t> inventory;
    std::map<Symbol, RiskMetrics> risk_metrics;
    
public:
    Quote adjust_quote_for_inventory(const Symbol& symbol,
                                   const Quote& base_quote) {
        int64_t current_inv = inventory[symbol];
        double inv_ratio = static_cast<double>(current_inv) / 
                          params.max_position;
        
        // Calculate inventory skew
        double skew = calculate_inventory_skew(inv_ratio);
        
        // Adjust quote prices
        Quote adjusted = base_quote;
        double mid = (base_quote.bid_price + base_quote.ask_price) / 2;
        double half_spread = (base_quote.ask_price - base_quote.bid_price) / 2;
        
        // Skew prices to reduce inventory
        adjusted.bid_price = mid - half_spread * (1 + skew);
        adjusted.ask_price = mid + half_spread * (1 - skew);
        
        // Adjust sizes based on inventory
        adjust_sizes_for_inventory(adjusted, inv_ratio);
        
        // Apply risk limits
        apply_risk_constraints(adjusted, symbol);
        
        return adjusted;
    }
    
private:
    double calculate_inventory_skew(double inv_ratio) {
        // Non-linear skew function optimized for Saudi market
        double base_skew = inv_ratio * params.inventory_skew_factor;
        
        // Increase skew exponentially as we approach limits
        double limit_factor = std::pow(std::abs(inv_ratio), 2);
        double total_skew = base_skew * (1 + limit_factor);
        
        // Cap maximum skew
        return std::clamp(total_skew, 
                         -params.max_skew_bps / 10000.0,
                         params.max_skew_bps / 10000.0);
    }
    
    void adjust_sizes_for_inventory(Quote& quote, double inv_ratio) {
        // Reduce size on the side that increases inventory
        if (inv_ratio > 0) {
            // Long inventory - reduce ask size
            quote.ask_size *= (1 - std::abs(inv_ratio));
            quote.ask_size = std::max(quote.ask_size, MIN_ORDER_SIZE);
        } else if (inv_ratio < 0) {
            // Short inventory - reduce bid size
            quote.bid_size *= (1 - std::abs(inv_ratio));
            quote.bid_size = std::max(quote.bid_size, MIN_ORDER_SIZE);
        }
        
        // Saudi market: maintain minimum presence
        quote.bid_size = std::max(quote.bid_size, 
                                  static_cast<uint32_t>(1000));
        quote.ask_size = std::max(quote.ask_size, 
                                  static_cast<uint32_t>(1000));
    }
    
    void apply_risk_constraints(Quote& quote, const Symbol& symbol) {
        auto& risk = risk_metrics[symbol];
        
        // Check VaR limits
        if (risk.current_var > risk.max_var * 0.8) {
            // Approaching VaR limit - reduce sizes
            quote.bid_size *= 0.5;
            quote.ask_size *= 0.5;
        }
        
        // Check inventory risk
        if (risk.inventory_risk > 0.1) {  // 10% portfolio risk
            // High inventory risk - widen spreads
            double adjustment = 1 + risk.inventory_risk;
            quote.bid_price /= adjustment;
            quote.ask_price *= adjustment;
        }
    }
};
```
