# 4.4 Statistical Market Making

```cpp
class StatisticalMarketMaker {
private:
    struct Feature {
        std::string name;
        double value;
        double weight;
    };
    
    struct MLModel {
        std::vector<Feature> features;
        std::function<double(const std::vector<Feature>&)> predictor;
        double accuracy;
        uint64_t last_update;
    };
    
    // Saudi-specific features
    class SaudiFeatureExtractor {
    public:
        std::vector<Feature> extract_features(
            const Symbol& symbol,
            const OrderBook& book,
            const MarketContext& context) {
            
            std::vector<Feature> features;
            
            // Microstructure features
            features.push_back({"spread_bps", book.get_spread_bps(), 1.0});
            features.push_back({"bid_ask_imbalance", 
                              book.get_imbalance_ratio(), 1.0});
            features.push_back({"book_depth_ratio", 
                              book.get_depth_ratio(), 1.0});
            
            // Saudi-specific features
            features.push_back({"minutes_to_prayer", 
                              get_minutes_to_next_prayer(), 1.0});
            features.push_back({"is_thursday", 
                              is_thursday() ? 1.0 : 0.0, 1.0});
            features.push_back({"sector_momentum", 
                              get_sector_momentum(symbol), 1.0});
            
            // Order flow features
            features.push_back({"order_flow_toxicity", 
                              calculate_flow_toxicity(symbol), 1.0});
            features.push_back({"retail_ratio", 
                              estimate_retail_ratio(book), 1.0});
            
            // Technical indicators adapted for Saudi market
            features.push_back({"rsi_5min", 
                              calculate_rsi(symbol, 5), 1.0});
            features.push_back({"volume_ratio", 
                              get_volume_ratio(symbol), 1.0});
            
            return features;
        }
        
    private:
        double estimate_retail_ratio(const OrderBook& book) {
            // Saudi market characteristic: small round lots indicate retail
            uint32_t retail_orders = 0;
            uint32_t total_orders = 0;
            
            for (const auto& level : book.bids) {
                if (is_retail_size(level.size)) {
                    retail_orders += level.order_count;
                }
                total_orders += level.order_count;
            }
            
            return static_cast<double>(retail_orders) / total_orders;
        }
        
        bool is_retail_size(uint32_t size) {
            // Common Saudi retail sizes
            return (size == 100 || size == 500 || size == 1000 || 
                    size == 5000 || size == 10000);
        }
    };
    
    SaudiFeatureExtractor feature_extractor;
    std::map<Symbol, MLModel> models;
    
public:
    Quote generate_statistical_quote(
        const Symbol& symbol,
        const OrderBook& book,
        const MarketContext& context) {
        
        // Extract features
        auto features = feature_extractor.extract_features(
            symbol, book, context);
        
        // Get model prediction
        auto& model = models[symbol];
        double prediction = model.predictor(features);
        
        // Convert prediction to quote adjustments
        Quote quote = generate_base_quote(symbol, book);
        apply_statistical_adjustments(quote, prediction, features);
        
        return quote;
    }
    
private:
    void apply_statistical_adjustments(
        Quote& quote,
        double prediction,
        const std::vector<Feature>& features) {
        
        // Prediction indicates expected price movement
        if (std::abs(prediction) > 0.001) {  // 0.1% threshold
            // Adjust spread based on directional confidence
            double spread_adjustment = 1.0 - std::abs(prediction) * 10;
            
            double mid = (quote.bid_price + quote.ask_price) / 2;
            double half_spread = (quote.ask_price - quote.bid_price) / 2;
            
            if (prediction > 0) {
                // Bullish - skew quotes up
                quote.bid_price = mid - half_spread * spread_adjustment;
                quote.ask_price = mid + half_spread * (2 - spread_adjustment);
            } else {
                // Bearish - skew quotes down
                quote.bid_price = mid - half_spread * (2 - spread_adjustment);
                quote.ask_price = mid + half_spread * spread_adjustment;
            }
        }
        
        // Adjust sizes based on confidence
        double confidence = calculate_prediction_confidence(features);
        quote.bid_size *= confidence;
        quote.ask_size *= confidence;
    }
};
```
