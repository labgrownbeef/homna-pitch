# 4.1 Passive Market Making

```cpp
class PassiveMarketMaker {
private:
    struct Configuration {
        // Saudi-specific parameters
        double base_spread_bps = 100;      // 1% base spread
        double min_spread_bps = 50;        // 0.5% minimum
        double max_spread_bps = 300;       // 3% maximum
        double prayer_spread_mult = 1.5;   // Widen during prayers
        
        // Size parameters
        uint32_t base_quote_size = 10000;  // Shares
        double size_skew_factor = 0.3;
        uint32_t min_quote_size = 1000;
        uint32_t max_quote_size = 50000;
        
        // Queue management
        uint32_t target_queue_position = 3;
        double queue_priority_threshold = 0.8;
    };
    
    Configuration config;
    std::unordered_map<Symbol, SymbolState> symbol_states;
    
public:
    Quote generate_quote(const Symbol& symbol, const OrderBook& book) {
        auto& state = symbol_states[symbol];
        
        // Calculate current spread based on market conditions
        double spread = calculate_dynamic_spread(symbol, book);
        
        // Get mid price
        double mid = book.get_mid_price();
        if (mid == 0) return Quote{};  // No market
        
        // Apply Saudi-specific adjustments
        spread = apply_saudi_adjustments(spread, get_current_time());
        
        // Generate quote prices
        Quote quote;
        quote.bid_price = round_to_tick(mid - spread / 2, symbol);
        quote.ask_price = round_to_tick(mid + spread / 2, symbol);
        
        // Calculate sizes based on inventory
        auto [bid_size, ask_size] = calculate_quote_sizes(symbol, state);
        quote.bid_size = bid_size;
        quote.ask_size = ask_size;
        
        // Queue position optimization
        optimize_queue_position(quote, book, state);
        
        return quote;
    }
    
private:
    double calculate_dynamic_spread(const Symbol& symbol, 
                                  const OrderBook& book) {
        // Base spread from configuration
        double spread_bps = config.base_spread_bps;
        
        // Adjust for current book conditions
        double book_spread = book.get_spread_bps();
        double imbalance = book.get_imbalance_ratio();
        
        // Volatility adjustment
        double volatility = calculate_realized_volatility(symbol);
        double vol_multiplier = 1.0 + (volatility / 0.02 - 1.0) * 0.5;
        
        // Saudi market: wider spreads acceptable
        spread_bps *= vol_multiplier;
        
        // Ensure we're competitive but profitable
        spread_bps = std::max(spread_bps, book_spread * 0.8);
        spread_bps = std::min(spread_bps, config.max_spread_bps);
        
        return spread_bps;
    }
    
    double apply_saudi_adjustments(double spread_bps, 
                                 const TimePoint& current_time) {
        // Check for prayer times
        if (is_dhuhr_time(current_time)) {
            spread_bps *= 1.5;  // 50% wider during Dhuhr
        } else if (is_asr_time(current_time)) {
            spread_bps *= 1.3;  // 30% wider during Asr
        }
        
        // Day of week adjustments (Thursday is last trading day)
        if (is_thursday(current_time)) {
            spread_bps *= 1.1;  // 10% wider on Thursdays
        }
        
        // Ramadan adjustments
        if (is_ramadan_period(current_time)) {
            spread_bps *= 1.2;  // 20% wider during Ramadan
        }
        
        return spread_bps;
    }
    
    void optimize_queue_position(Quote& quote, 
                               const OrderBook& book,
                               SymbolState& state) {
        // Saudi market: queue position more valuable
        auto bid_queue_depth = book.get_queue_depth(quote.bid_price);
        auto ask_queue_depth = book.get_queue_depth(quote.ask_price);
        
        // If queue is too deep, improve price
        if (bid_queue_depth > config.target_queue_position * 2) {
            quote.bid_price += get_tick_size(quote.bid_price);
        }
        
        if (ask_queue_depth > config.target_queue_position * 2) {
            quote.ask_price -= get_tick_size(quote.ask_price);
        }
    }
};
```
