# 5.1 Prayer Time Management

```python
class PrayerTimeManager:
    def __init__(self):
        self.prayer_times = self.load_prayer_times()
        
    def get_trading_adjustments(self, current_time):
        """
        Returns trading parameter adjustments for prayer times
        """
        adjustments = {
            'spread_multiplier': 1.0,
            'size_multiplier': 1.0,
            'cancel_threshold': 1.0,
            'risk_multiplier': 1.0
        }
        
        # Check Dhuhr (Noon prayer) - 12:00-12:30
        if self.is_dhuhr_time(current_time):
            adjustments['spread_multiplier'] = 1.5     # 50% wider spreads
            adjustments['size_multiplier'] = 0.6       # 40% smaller sizes
            adjustments['cancel_threshold'] = 0.7      # More aggressive cancels
            adjustments['risk_multiplier'] = 0.5       # Reduce risk
            
        # Check Asr (Afternoon prayer) - overlaps with close
        elif self.is_asr_time(current_time):
            adjustments['spread_multiplier'] = 1.3     # 30% wider spreads
            adjustments['size_multiplier'] = 0.7       # 30% smaller sizes
            adjustments['cancel_threshold'] = 0.8      # Moderate cancels
            adjustments['risk_multiplier'] = 0.7       # Reduce risk
            
        return adjustments
        
    def get_liquidity_forecast(self, symbol, time_horizon_minutes=30):
        """
        Forecast liquidity changes due to prayer times
        """
        current_time = datetime.now()
        forecast = []
        
        for minutes in range(0, time_horizon_minutes, 5):
            future_time = current_time + timedelta(minutes=minutes)
            
            # Base liquidity
            liquidity = 1.0
            
            # Adjust for prayer times
            if self.is_dhuhr_time(future_time):
                liquidity *= 0.6  # 40% reduction
            elif self.is_asr_time(future_time):
                liquidity *= 0.7  # 30% reduction
                
            # Pre-prayer surge (5-10 minutes before)
            minutes_to_dhuhr = self.minutes_until_dhuhr(future_time)
            if 5 <= minutes_to_dhuhr <= 10:
                liquidity *= 1.2  # 20% increase as traders close positions
                
            forecast.append({
                'time': future_time,
                'liquidity_factor': liquidity,
                'is_prayer_time': liquidity < 1.0
            })
            
        return forecast
```
