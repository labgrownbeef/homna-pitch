# 5.2 Tick Size Optimization

```cpp
class SaudiTickOptimizer {
private:
    struct TickLevel {
        double price_threshold;
        double tick_size;
    };
    
    // Tadawul tick size structure
    const std::vector<TickLevel> tick_structure = {
        {0.00,   0.01},   // Below 50 SAR: 0.01 tick
        {50.00,  0.05},   // 50-100 SAR: 0.05 tick  
        {100.00, 0.10}    // Above 100 SAR: 0.10 tick
    };
    
public:
    double optimize_quote_price(double raw_price, Side side) {
        double tick = get_tick_size(raw_price);
        
        if (side == Side::BID) {
            // For bids, use aggressive rounding to improve queue position
            return floor(raw_price / tick) * tick;
        } else {
            // For asks, use conservative rounding
            return ceil(raw_price / tick) * tick;
        }
    }
    
    double calculate_tick_advantage(double price, int queue_position) {
        // Value of being at a price level in Saudi market
        double tick = get_tick_size(price);
        
        // Each tick is worth more in Saudi due to wider spreads
        double tick_value_bps = (tick / price) * 10000;
        
        // Queue position value (exponential decay)
        double queue_factor = exp(-queue_position / 5.0);
        
        return tick_value_bps * queue_factor;
    }
    
private:
    double get_tick_size(double price) {
        for (auto it = tick_structure.rbegin(); 
             it != tick_structure.rend(); ++it) {
            if (price >= it->price_threshold) {
                return it->tick_size;
            }
        }
        return 0.01;  // Default
    }
};
```
