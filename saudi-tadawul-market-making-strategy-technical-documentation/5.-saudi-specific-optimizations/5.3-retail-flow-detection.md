# 5.3 Retail Flow Detection

```python
class RetailFlowAnalyzer:
    def __init__(self):
        self.retail_patterns = {
            'round_lots': [100, 500, 1000, 5000, 10000],
            'typical_order_values': [10000, 25000, 50000, 100000],  # SAR
            'common_strategies': ['market_orders', 'stop_losses', 'round_prices']
        }
        
    def analyze_order_flow(self, order_stream):
        """
        Identify retail vs institutional flow patterns
        """
        retail_score = 0.0
        total_orders = len(order_stream)
        
        for order in order_stream:
            # Check for round lot sizes
            if order.size in self.retail_patterns['round_lots']:
                retail_score += 1.0
                
            # Check for round price levels
            if self.is_round_price(order.price):
                retail_score += 0.5
                
            # Check for market orders (retail tendency)
            if order.type == 'MARKET':
                retail_score += 0.8
                
            # Check order value
            order_value = order.size * order.price
            if order_value in self.retail_patterns['typical_order_values']:
                retail_score += 0.7
                
        return retail_score / total_orders
        
    def predict_retail_behavior(self, market_state):
        """
        Predict retail trading patterns based on market conditions
        """
        predictions = {}
        
        # Retail tends to chase momentum
        if market_state.momentum > 0.02:  # 2% move
            predictions['retail_buying_pressure'] = 0.8
            predictions['expected_market_orders'] = 0.6
            
        # Retail panic sells on drops
        elif market_state.momentum < -0.02:
            predictions['retail_selling_pressure'] = 0.9
            predictions['stop_loss_triggers'] = 0.7
            
        # Retail less active during prayer times
        if market_state.is_prayer_time:
            predictions['retail_activity'] = 0.3
        else:
            predictions['retail_activity'] = 0.8
            
        return predictions
```
