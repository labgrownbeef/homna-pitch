# 2.2 Strategy Allocation Framework

```python
class StrategyAllocator:
    def __init__(self):
        self.base_allocations = {
            'passive': 0.40,      # 40% - Consistent spread capture
            'inventory': 0.35,    # 35% - Dynamic inventory management
            'aggressive': 0.15,   # 15% - Opportunistic trading
            'statistical': 0.10   # 10% - ML-driven decisions
        }
        
    def calculate_dynamic_allocation(self, market_state):
        """
        Adjust allocations based on real-time market conditions
        """
        allocations = self.base_allocations.copy()
        
        # Adjust for volatility
        if market_state.volatility > market_state.avg_volatility * 1.5:
            # High volatility: reduce passive, increase aggressive
            allocations['passive'] *= 0.7
            allocations['aggressive'] *= 1.5
            
        # Adjust for prayer times
        if market_state.is_prayer_time:
            # Reduce all strategies except statistical
            for strategy in ['passive', 'inventory', 'aggressive']:
                allocations[strategy] *= 0.6
            allocations['statistical'] *= 1.2
            
        # Normalize allocations
        total = sum(allocations.values())
        return {k: v/total for k, v in allocations.items()}
```
